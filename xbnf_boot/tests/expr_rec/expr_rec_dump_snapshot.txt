       / atom

prefix = ("+" / "-") suffix
       / suffix

# FIRST = {"(", "+", "-", IDENT, INT, STR}
mul = mul ("*" / "/") prefix
    / prefix

# FIRST = {"(", "+", "-", IDENT, INT, STR}
add = add ("+" / "-") mul
    / mul
